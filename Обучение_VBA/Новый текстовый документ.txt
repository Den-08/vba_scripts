Private Sub Workbook_Open()
Select Case Time
    Case Is < 0.5, Is > 0.25
        MsgBox "Доброе утро!"
    Case Is < 0.75, Is > 0.5
        MsgBox "Добрый день!"
    Case Is < 1, Is > 0.75
        MsgBox "Добрый вечер!"
    Case Is < 1, Is < 0.25
        MsgBox "Доброй ночи!"
End Select
End Sub


Private Sub Workbook_Open()
'Time возвращает время которое является долей от единицы суток
Select Case Time
    Case 7 / 24 To 12 / 24
        MsgBox ("доброе утро")
    Case 12 / 24 To 18 / 24
        MsgBox ("добрый день")
    Case 18 / 24 To 23 / 24
        MsgBox ("добрый вечер")
    Case 23 / 24 To 24 / 24, 0 / 24 To 7 / 24
        MsgBox ("доброй ночи")
End Select
End Sub
